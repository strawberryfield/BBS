139  public virtual void HandleChar(byte[] data, int bytesReceived)
140  {
141  switch (data[0])
142  {
143  case 0x03:
144  HandleControlC();
145  break;
146  case 0x09:
147  HandleTab();
148  break;
149  case 0x1b: // ESC
150  if (bytesReceived == 1) HandleESC();
151  else if (bytesReceived == 3 && data[1] == '[')
152  {
153  switch (data[2])
154  {
155  case 65:
156  HandleCursorUp();
157  break;
158  case 66:
159  HandleCursorDown();
160  break;
161  case 67:
162  HandleCursorRight();
163  break;
164  case 68:
165  HandleCursorLeft();
166  break;
167  default:
168  break;
169  }
170  }
171  else if (bytesReceived == 4 && data[1] == '[' && data[3] == 126)
172  {
173  switch (data[2])
174  {
175  case 53:
176  HandlePageUp();
177  break;
178  case 54:
179  HandlePageDown();
180  break;
181  case 49:
182  HandleHome();
183  break;
184  case 52:
185  HandleEnd();
186  break;
187  default:
188  break;
189  }
190  }
191  else if (bytesReceived == 3 && data[1] == 'O')
192  {
193  switch (data[2])
194  {
195  case 80:
196  HandleF1();
197  break;
198  case 81:
199  HandleF2();
200  break;
201  case 82:
202  HandleF3();
203  break;
204  case 83:
205  HandleF4();
206  break;
207  default:
208  break;
209  }
210  }
211  break;
212  default:
213  break;
214  }
