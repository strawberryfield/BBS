75  public override void EnterBbsCodeElement([NotNull] BBSCodeParser.BbsCodeElementContext context)
76  {
77  string tagName = context.children[1].GetText().Trim().ToUpper();
78  Tags tag;
79  if (TagsTable.TryGetValue(tagName, out tag))
80  {
81  Parsed.TextPush();
82  switch (tag)
83  {
84  case Tags.BLINK:
85  ANSI.SetMode(ANSICodes.Modes.Blink);
86  Parsed.TextConcat(ANSI.WriteMode());
87  break;
88  case Tags.REVERSE:
89  ANSI.SetMode(ANSICodes.Modes.Reverse);
90  Parsed.TextConcat(ANSI.WriteMode());
91  break;
92  case Tags.BOLD:
93  ANSI.SetMode(ANSICodes.Modes.Bold);
94  Parsed.TextConcat(ANSI.WriteMode());
95  break;
96  case Tags.UNDERLINE:
97  ANSI.SetMode(ANSICodes.Modes.Underline);
98  Parsed.TextConcat(ANSI.WriteMode());
99  break;
100  case Tags.HEADER:
101  case Tags.FOOTER:
102  case Tags.BODY:
103  case Tags.HIDDEN:
104  Parsed.TextClear();
105  break;
106  default:
107  break;
108  }
109  }
110  }

116  public override void ExitBbsCodeElement([NotNull] BBSCodeParser.BbsCodeElementContext context)
117  {
118  string tagName = context.children[1].GetText().Trim().ToUpper();
119  Tags tag;
120  if (TagsTable.TryGetValue(tagName, out tag))
121  {
122  Attributes attr = AttributesTable.GetAttributes(tag);
123  string value;
124  string value1;
125  switch (tag)
126  {
127  case Tags.CLS:
128  if (attr.TryGetValue("FORECOLOR", out value)) ANSI.pushForeColor(value);
129  if (attr.TryGetValue("BACKCOLOR", out value))
130  {
131  ANSI.pushBackColor(value);
132  }
133  Parsed.TextConcat(ANSI.ClearScreen());
134  Parsed.TextPop(true);
135  break;
136  case Tags.BLINK:
137  ANSI.ResetMode(ANSICodes.Modes.Blink);
138  Parsed.TextConcat(ANSI.WriteMode());
139  Parsed.TextPop(true);
140  break;
141  case Tags.REVERSE:
142  ANSI.ResetMode(ANSICodes.Modes.Reverse);
143  Parsed.TextConcat(ANSI.WriteMode());
144  Parsed.TextPop(true);
145  break;
146  case Tags.BOLD:
147  ANSI.ResetMode(ANSICodes.Modes.Bold);
148  Parsed.TextConcat(ANSI.WriteMode());
149  Parsed.TextPop(true);
150  break;
151  case Tags.UNDERLINE:
152  ANSI.ResetMode(ANSICodes.Modes.Underline);
153  Parsed.TextConcat(ANSI.WriteMode());
154  Parsed.TextPop(true);
155  break;
156  case Tags.COLOR:
157  ANSI.popForeColor();
158  Parsed.TextConcat(ANSI.WriteMode());
159  Parsed.TextPop(true);
160  break;
161  case Tags.BACKCOLOR:
162  ANSI.popBackColor();
163  Parsed.TextConcat(ANSI.WriteMode());
164  Parsed.TextPop(true);
165  break;
166  case Tags.P:
167  Parsed.TextAssign(string.Join('\n',
168  TextHelper.WordWrap(Parsed.Parsed,
169  Client.screenWidth).ToArray()));
170  Parsed.TextPop(true);
171  break;
172  case Tags.MOVE:
173  if (!attr.TryGetValue("ROW", out value)) value = "0";
174  if (!attr.TryGetValue("COL", out value1)) value1 = "0";
175  Parsed.TextConcat(ANSI.Move(value1, value));
176  Parsed.TextPop(true);
177  break;
178  case Tags.FIGGLE:
179  if (!attr.TryGetValue("FONT", out value)) value = "standard";
180  try
181  {
182  Parsed.TextAssign(Figgle.FiggleFonts.Lookup(value.ToLower()).Render(Parsed.Parsed));
183  }
184  catch (Exception)
185  { }
186  Parsed.TextPop(true);
187  break;
188  case Tags.ACTION:
189  action = new BBSCodeResult.Action(attr);
190  if (!string.IsNullOrWhiteSpace(action.requires))
191  {
192  if (string.IsNullOrWhiteSpace(Client.username))
193  {
194  Parsed.TextPop(true);
195  return;
196  }
197  using (bbsContext bbs = new bbsContext())
198  {
199  User user = bbs.GetUserByUsername(Client.username);
200  bool allowed = user.HasRights(action.requires);
201  Parsed.TextPop(allowed);
202  if (!allowed) return;
203  }
204  }
205  else Parsed.TextPop(true);
206  if (!Parsed.Actions.TryAdd(action.key, action))
207  EventLogger.Write(string.Format("Error adding action '{0}' in '{1}'", action.key, FileName), 0);
208  break;
209  case Tags.BEEP:
210  Parsed.TextConcat((char)7);
211  Parsed.TextPop(true);
212  break;
213  case Tags.HR:
214  Parsed.TextConcat(TextHelper.HR('-', Client.screenWidth));
215  Parsed.TextPop(true);
216  break;
217  case Tags.HEADER:
218  if (attr.TryGetValue("BACKCOLOR", out value))
219  Parsed.HeaderBackground = ANSI.GetColorByName(value, true);
220  Parsed.Header = Parsed.Parsed;
221  Parsed.TextClear();
222  break;
223  case Tags.FOOTER:
224  if (attr.TryGetValue("BACKCOLOR", out value))
225  Parsed.FooterBackground = ANSI.GetColorByName(value, true);
226  Parsed.Footer = Parsed.Parsed;
227  Parsed.TextClear();
228  break;
229  case Tags.BODY:
230  if (attr.TryGetValue("ALTERNATECOLOR", out value))
231  Parsed.BodyAlternateBackground = ANSI.GetColorByName(value, true);
232  if (attr.TryGetValue("FOCUSEDCOLOR", out value))
233  Parsed.BodyFocusedBackground = ANSI.GetColorByName(value, true);
234  Parsed.Body = Parsed.Parsed;
235  Parsed.TextClear();
236  break;
237  case Tags.HIDDEN:
238  Parsed.TextClear();
239  break;
240  case Tags.H1:
241  Parsed.TextAssign(string.Join('\n',
242  ANSI.Header(Parsed.Parsed, 1,
243  Client.screenWidth).ToArray()));
244  Parsed.TextPop(true);
245  break;
246  case Tags.H2:
247  Parsed.TextAssign(string.Join('\n',
248  ANSI.Header(Parsed.Parsed, 2,
249  Client.screenWidth).ToArray()));
250  Parsed.TextPop(true);
251  break;
252  case Tags.H3:
253  Parsed.TextAssign(string.Join('\n',
254  ANSI.Header(Parsed.Parsed, 3,
255  Client.screenWidth).ToArray()));
256  Parsed.TextPop(true);
257  break;
258  case Tags.H4:
259  Parsed.TextAssign(string.Join('\n',
260  ANSI.Header(Parsed.Parsed, 1,
261  Client.screenWidth).ToArray()));
262  Parsed.TextPop(true);
263  break;
264  default:
265  break;
266  }
267  }

274  public override void EnterBbsCodeAttribute([NotNull] BBSCodeParser.BbsCodeAttributeContext context)
275  {
276  string tagName = context.Parent.GetChild(1).GetText().Trim().ToUpper();
277  string attributeName = context.children[0].GetText().Trim().ToUpper();
278  string attributeValue = string.Empty;
279  if(context.ChildCount > 2) attributeValue = context.children[2].GetText().Trim('"').Trim();
280 
281  Tags tag;
282  if (TagsTable.TryGetValue(tagName, out tag))
283  {
284  switch (tag)
285  {
286  case Tags.COLOR:
287  ANSI.pushForeColor(attributeValue);
288  Parsed.TextConcat(ANSI.WriteMode());
289  break;
290  case Tags.BACKCOLOR:
291  ANSI.pushBackColor(attributeValue);
292  Parsed.TextConcat(ANSI.WriteMode());
293  break;
294  default:
295  AttributesTable.Add(tag, attributeName, attributeValue);
296  break;
297  }
298  }
299  }
