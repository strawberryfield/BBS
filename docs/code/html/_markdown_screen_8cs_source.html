129  protected List<string> BlockProcessing(MarkdownBlock b, int recursionLevel)
130  {
131  string forecolor = string.Empty;
132  string backcolor = string.Empty;
133  string underline = string.Empty;
134  int maxWidth = client.screenWidth - recursionLevel * 2;
135  List<string> Text = new List<string>();
136  switch (b.Type)
137  {
138  case MarkdownBlockType.Root:
139  break;
140  case MarkdownBlockType.Paragraph:
141  string para = "";
142  foreach (var i in ((ParagraphBlock)b).Inlines)
143  para += ProcessInline(i);
144  Text.AddRange(TextHelper.WordWrap(para, maxWidth));
145  Text.Add(string.Empty);
146  break;
147  case MarkdownBlockType.Quote:
148  break;
149  case MarkdownBlockType.Code:
150  forecolor = ANSI.GetStyleForeColor("Code_Color", mdStyles);
151  backcolor = ANSI.GetStyleBackColor("Code_Back", mdStyles);
152  List<string> rows = TextHelper.SplitString(((CodeBlock)b).Text);
153  foreach (string s in rows)
154  {
155  Text.Add(backcolor + forecolor + ANSI.ClearCurrentLine + s + ANSI.WriteMode());
156  }
157  Text.Add(ANSI.WriteMode());
158  break;
159  case MarkdownBlockType.Header:
160  HeaderBlock hb = (HeaderBlock)b;
161  Text.AddRange(ANSI.Header(hb.ToString(), hb.HeaderLevel, maxWidth));
162  break;
163  case MarkdownBlockType.List:
164  ListBlock lb = (ListBlock)b;
165  string indent = new string(' ', recursionLevel * 2);
166  foreach (var li in lb.Items)
167  {
168  List<string> lir = new List<string>();
169  bool isFirst = true;
170  foreach (var lib in li.Blocks)
171  lir.AddRange(BlockProcessing(lib, recursionLevel + 1));
172  foreach (string s in lir)
173  {
174  Text.Add((isFirst ? "- " : " ") + s);
175  isFirst = false;
176  }
177  }
178  break;
179  case MarkdownBlockType.ListItemBuilder:
180  break;
181  case MarkdownBlockType.HorizontalRule:
182  Text.Add(TextHelper.HR('-', maxWidth));
183  break;
184  case MarkdownBlockType.Table:
185  TableBlock tb = (TableBlock)b;
186  int colWidth = maxWidth / tb.ColumnDefinitions.Count;
187  foreach(var tr in tb.Rows)
188  {
189  string trs = string.Empty;
190  foreach(var tc in tr.Cells)
191  {
192  string tcs = string.Empty;
193  foreach (var i in tc.Inlines)
194  tcs += ProcessInline(i);
195  trs += TextHelper.Truncate(tcs, colWidth - 1).PadRight(colWidth);
196  }
197  Text.Add(trs);
198  }
199  break;
200  case MarkdownBlockType.LinkReference:
201  break;
202  case MarkdownBlockType.YamlHeader:
203  break;
204  default:
205  break;
206  }
207  return Text;
208  }
