103  private const byte ModeNormal = 0b00000000;
104  private const byte ModeBold = 0b00000010;
105  private const byte ModeUnderline = 0b00000100;
106  private const byte ModeBlink = 0b00001000;
107  private const byte ModeReverse = 0b00010000;

109  private string ModeFromBits(byte status)
110  {
111  string ret = "0;";
112  if ((status & ModeBold) != 0) ret += string.Format("{0};", (int)Modes.Bold);
113  if ((status & ModeUnderline) != 0) ret += string.Format("{0};", (int)Modes.Underline);
114  if ((status & ModeBlink) != 0) ret += string.Format("{0};", (int)Modes.Blink);
115  if ((status & ModeReverse) != 0) ret += string.Format("{0};", (int)Modes.Reverse);
116  return ret;
117  }
118 
119  private byte BitsFromMode(Modes m)
120  {
121  if (m == Modes.Bold) return ModeBold;
122  if (m == Modes.Underline) return ModeUnderline;
123  if (m == Modes.Blink) return ModeBlink;
124  if (m == Modes.Reverse) return ModeReverse;
125  return ModeNormal;
126  }

380  public List<string> Header(string s, int level, int width)
381  {
382  List<string> Text = new List<string>();
383  string l = string.Format("H{0}_", level);
384  string forecolor = GetStyleForeColor(l + "Color");
385  string backcolor = GetStyleBackColor(l + "Back");
386  string underline = appearance.Get(l + "Underline");
387  foreach (string str in TextHelper.WordWrap(s, width))
388  {
389  Text.Add(backcolor + forecolor + ClearCurrentLine + str + WriteMode());
390  }
391  if (!string.IsNullOrWhiteSpace(underline))
392  Text.Add(backcolor + forecolor + TextHelper.HR(underline[0], width) + WriteMode());
393  Text.Add(WriteMode());
394  return Text;
395  }
