85  public override void HandleMessage(string msg)
86  {
87  msg = msg.Trim();
88  if (string.IsNullOrWhiteSpace(msg) && status != states.WaitForContinue) return;
89  switch (status)
90  {
91  case states.WaitForUsername:
92  msg = msg.ToUpper();
93  if(msg == "EXIT")
94  {
95  ShowNext();
96  return;
97  }
98  using (bbsContext bbs = new bbsContext())
99  user = bbs.GetUserByUsername(msg);
100  if(user == null)
101  {
102  LnWrite(catalog.GetString("User '{0}' is not found.", msg));
103  LnWrite(catalog.GetString("Username") + ": ");
104  }
105  else
106  {
107  LnWrite(catalog.GetString("New password") + ": ");
108  client.status = EClientStatus.Authenticating;
109  status = states.WaitForNewPassword;
110  }
111  break;
112  case states.WaitForNewPassword:
113  handleWaitForNewPassword(msg);
114  break;
115  case states.WaitForConfirm:
116  handleWaitForConfirm(msg);
117  break;
118  case states.WaitForContinue:
119  ShowNext();
120  break;
121  default:
122  break;
123  }
124  }
