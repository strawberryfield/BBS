82  static public string LineAlign(string text, int width, TextAlign align)
83  {
84  text = text.Trim();
85  if (text.Length >= width) return text;
86  switch (align)
87  {
88  case TextAlign.Left:
89  return text;
90  case TextAlign.Right:
91  return text.PadLeft(width);
92  case TextAlign.Center:
93  int leftPadding = (width - text.Length) / 2;
94  int rightPadding = width - text.Length - leftPadding;
95  return new string(' ', leftPadding) + text + new string(' ', rightPadding);
96  case TextAlign.Both:
97  int missingSpaces = width - text.Length;
98  string[] words = text.Split(' ');
99  int breaks = words.Length - 1;
100  int quotient = missingSpaces / breaks;
101  int reminder = missingSpaces % breaks;
102  string ret = string.Empty;
103  for (int j = 0; j < breaks; j++)
104  ret += words[j] + new string(' ', 1 + quotient + (j < reminder ? 1 : 0));
105  return ret + words[breaks];
106  default:
107  return text;
108  }
109  }
