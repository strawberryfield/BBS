103  public class EntitiesDict
104  {
105  private Dictionary<string, Entities> EntitiesTable;
106  private Dictionary<string, string> CustomEntitiesTable;
107 
108  private IBBSClient client;
109 
114  public EntitiesDict(IBBSClient c)
115  {
116  client = c;
117 
118  EntitiesTable = new Dictionary<string, Entities>();
119  foreach (Entities t in Enum.GetValues(typeof(Entities)))
120  EntitiesTable.Add(t.ToString().ToUpper(), t);
121 
122  CustomEntitiesTable = new Dictionary<string, string>();
123  NameValueCollection config = (NameValueCollection)ConfigurationManager.GetSection("Entities");
124  foreach (string key in config)
125  CustomEntitiesTable.Add(key.Trim().ToUpper(), config[key]);
126  }
127 
133  public string GetValue(string name)
134  {
135  Entities entity;
136  if (EntitiesTable.TryGetValue(name, out entity)) switch (entity)
137  {
138  case Entities.AMP:
139  return "&";
140  case Entities.LEFTCURLY:
141  return "{";
142  case Entities.RIGHTCURLY:
143  return "}";
144  case Entities.USERNAME:
145  return string.IsNullOrWhiteSpace(client.username) ? "GUEST" : client.username;
146  case Entities.REMOTE:
147  return client.Remote;
148  case Entities.CONNECTIONTIME:
149  return client.connectedAt.ToString("g");
150  case Entities.AGRAVE:
151  return "a'";
152  case Entities.EGRAVE:
153  return "e'";
154  case Entities.IGRAVE:
155  return "i'";
156  case Entities.OGRAVE:
157  return "o'";
158  case Entities.UGRAVE:
159  return "u'";
160  case Entities.EACUTE:
161  return "e'";
162  case Entities.SCREENWIDTH:
163  return client.screenWidth.ToString();
164  case Entities.SCREENHEIGHT:
165  return client.screenHeight.ToString();
166  case Entities.TERMINALTYPE:
167  return client.terminalType;
168  case Entities.BITS:
169  return client.BinaryMode ? "8" : "7";
170  default:
171  return string.Empty;
172  }
173  else
174  {
175  string customEntity;
176  if (CustomEntitiesTable.TryGetValue(name, out customEntity))
177  return customEntity;
178  else
179  return string.Empty;
180  }
181 
182  }
183  }
