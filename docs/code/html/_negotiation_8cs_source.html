390  public static void HandleWindowSize(byte[] data, int offset, IBBSClient c)
391  {

461  public static bool HandleTerminalType(byte[] data, int offset, IBBSClient c)
462  {
463  if (data[offset + 1] == (byte)Tokens.Subnegotiation
464  && data[offset + 2] == (byte)Operations.TerminalType
465  && data[offset + 3] == 0)
466  {
467  StringBuilder sb = new StringBuilder();
468  for (int j = offset + 4; data[j] != (byte)Tokens.IAC; j++)
469  sb.Append((char)data[j]);
470 
471  if (string.IsNullOrWhiteSpace(c.terminalType))
472  c.terminalType = sb.ToString();
473 
474  return c.TryAddTerminalType(sb.ToString());
475  }
476  return false;
477  }
