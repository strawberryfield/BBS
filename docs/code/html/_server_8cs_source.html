278  private void handleIncomingConnection(IAsyncResult result)
279  {
280  try
281  {
282  Socket oldSocket = (Socket)result.AsyncState;
283 
284  if (acceptIncomingConnections)
285  {
286  Socket newSocket = oldSocket.EndAccept(result);
287 
288  uint clientID = (uint)clients.Count + 1;
289  Client client = new Client(clientID, (IPEndPoint)newSocket.RemoteEndPoint);
290  clients.Add(newSocket, client);
291 
292  // Start negotiation
293  sendBytesToSocket(newSocket, Negotiation.Do(Negotiation.Operations.Echo));
294  sendBytesToSocket(newSocket, Negotiation.Do(Negotiation.Operations.RemoteFlowControl));
295  sendBytesToSocket(newSocket, Negotiation.Will(Negotiation.Operations.Echo));
296  sendBytesToSocket(newSocket, Negotiation.Will(Negotiation.Operations.SuppressGoAhead));
297  sendBytesToSocket(newSocket, Negotiation.Will(Negotiation.Operations.BinaryTransmission));
298  sendBytesToSocket(newSocket, Negotiation.Do(Negotiation.Operations.NegotiateAboutWindowSize));
299  sendBytesToSocket(newSocket, Negotiation.Do(Negotiation.Operations.TerminalType));
300 
301  client.resetReceivedData();
302 
303  ClientConnected(client);
304 
305  serverSocket.BeginAccept(new AsyncCallback(handleIncomingConnection), serverSocket);
306  }
307 
308  else
309  {
310  ConnectionBlocked((IPEndPoint)oldSocket.RemoteEndPoint);
311  }
312  }

339  private void receiveData(IAsyncResult result)
340  {
341  try
342  {
343  Socket clientSocket = (Socket)result.AsyncState;
344  Client client = getClientBySocket(clientSocket);
345 
346  int bytesReceived = clientSocket.EndReceive(result);
347 
348  if (bytesReceived == 0)
349  {
350  closeSocket(clientSocket);
351  ClientDisconnected(client);
352  serverSocket.BeginAccept(new AsyncCallback(handleIncomingConnection), serverSocket);
353  }
354  else if (data[0] == (byte)Negotiation.Tokens.IAC)
355  {
356  // Negotiation
357  int offset = 0;
358  while (offset < bytesReceived)
359  {
360  Negotiation.HandleBinary(data, offset, client);
361  Negotiation.HandleWindowSize(data, offset, client);
362  if (Negotiation.ClientWillTerminalType(data, offset))
363  sendBytesToSocket(clientSocket, Negotiation.AskForTerminalType());
364  if (Negotiation.HandleTerminalType(data, offset, client))
365  sendBytesToSocket(clientSocket, Negotiation.AskForTerminalType());
366  while(++offset < bytesReceived)
367  {
368  if (data[offset] == (byte)Negotiation.Tokens.IAC) break;
369  }
370 
371  }
372  clientSocket.BeginReceive(data, 0, dataSize, SocketFlags.None, new AsyncCallback(receiveData), clientSocket);
373  }
374  else if (data[0] < 0xF0)
375  {
376  string receivedData = client.receivedData;
377  client.lastActivity = DateTime.Now;
378 
379  // 0x2E = '.', 0x0D = carriage return, 0x0A = new line
380  if ((data[0] == 0x2E && data[1] == 0x0D && receivedData.Length == 0) ||
381  (data[0] == 0x0D && data[1] == 0x0A) || data[0] == 0x0D || data[0] == 0x0A)
382  {
383  //sendMessageToSocket(clientSocket, "\u001B[1J\u001B[H");
384  MessageReceived(client, client.receivedData);
385  client.resetReceivedData();
386  }
387 
388  else
389  {
390  // 0x08 => backspace character
391  if (data[0] == 0x08)
392  {
393  if (receivedData.Length > 0)
394  {
395  client.removeLastCharacterReceived();
396  sendBytesToSocket(clientSocket, new byte[] { 0x08, 0x20, 0x08 });
397  }
398 
399  else
400  clientSocket.BeginReceive(data, 0, dataSize, SocketFlags.None, new AsyncCallback(receiveData), clientSocket);
401  }
402 
403  // 0x7F => delete character
404  else if (data[0] == 0x7F)
405  clientSocket.BeginReceive(data, 0, dataSize, SocketFlags.None, new AsyncCallback(receiveData), clientSocket);
406  else if (data[0] < 0x20)
407  ControlCharReceived(client, data, bytesReceived);
408  else
409  {
410  client.appendReceivedData(Encoding.ASCII.GetString(data, 0, bytesReceived));
411 
412  // Echo back the received character
413  // if client is not writing any password
414  if (client.status != EClientStatus.Authenticating)
415  sendBytesToSocket(clientSocket, new byte[] { data[0] });
416 
417  // Echo back asterisks if client is
418  // writing a password
419  else
420  sendMessageToSocket(clientSocket, "*");
421 
422  clientSocket.BeginReceive(data, 0, dataSize, SocketFlags.None, new AsyncCallback(receiveData), clientSocket);
423  }
424  }
425  }
426 
427  else
428  clientSocket.BeginReceive(data, 0, dataSize, SocketFlags.None, new AsyncCallback(receiveData), clientSocket);
429  }
430 
431  catch { }
432  }
