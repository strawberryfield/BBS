78  public bool AcceptablePassword(string pwd)
79  {
80  NameValueCollection opt = GetSecurityOptions();
81 
82  if (pwd.Length < Convert.ToInt16(opt["PasswordMinLength"])) return false;
83  if(opt["PasswordWantUpperLower"].ToUpper()[0] == 'Y')
84  {
85  if (!pwd.Any(char.IsUpper) || !pwd.Any(char.IsLower)) return false;
86  }
87  if (opt["PasswordWantDigit"].ToUpper()[0] == 'Y')
88  {
89  if (!pwd.Any(char.IsDigit)) return false;
90  }
91  if (opt["PasswordWantSpecial"].ToUpper()[0] == 'Y')
92  {
93  if (!pwd.Any(c => !char.IsLetterOrDigit(c))) return false;
94  }
95  return true;
96  }
