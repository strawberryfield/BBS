101  public override void HandleMessage(string msg)
102  {
103  switch (status)
104  {
105  case states.WaitForUsername:
106  username = msg.Trim().ToUpper();
107  if (username == string.Empty) break;
108  switch (username)
109  {
110  case "GUEST":
111  client.screen = ScreenFactory.Create(client, server, "TextScreen", "@GuestAccess");
112  client.screen.Show();
113  break;
114  case "NEW":
115  client.screen = ScreenFactory.Create(client, server, "NewUser");
116  client.screen.Show();
117  break;
118  case "LOGOUT":
119  client.screen = ScreenFactory.Create(client, server, "Logout");
120  client.screen.Show();
121  break;
122  default:
123  LnWrite(catalog.GetString("Password") + ": ");
124  status = states.WaitForPassword;
125  client.status = EClientStatus.Authenticating;
126  break;
127  }
128  break;
129  case states.WaitForPassword:
130  string pwd = msg.Trim();
131  client.status = EClientStatus.Guest;
132 
133  bool success = false;
134  using (bbsContext bbs = new bbsContext())
135  {
136  User user = bbs.GetUserByUsername(username);
137  if (user != null)
138  {
139  Login login = new Login() { UserId = username, From = client.Remote };
140  if (user.CheckPassword(pwd))
141  {
142  // successful login
143  success = true;
144  user.LastLoginDate = DateTime.Now;
145  user.LastLoginFrom = client.Remote;
146  login.Success = true;
147  bbs.Logins.Add(login);
148  try
149  {
150  bbs.SaveChanges();
151  }
152  catch (Exception e)
153  {
154  Console.WriteLine(e.Message);
155  Console.WriteLine(e.InnerException.Message);
156  }
157 
158  EventLogger.Write(string.Format("Successful login for user '{0}'", username),
159  client.Remote);
160  client.status = EClientStatus.LoggedIn;
161  if (string.IsNullOrWhiteSpace(client.locale)) client.locale = user.Locale;
162  }
163  else
164  {
165  login.Success = false;
166  bbs.Logins.Add(login);
167  bbs.SaveChanges();
168  EventLogger.Write(string.Format("Password failed for user '{0}'", username),
169  client.Remote);
170  }
171  }
172  else
173  {
174  EventLogger.Write(string.Format("Unknown user '{0}'", username),
175  client.Remote);
176  }
177  }
178 
179  if (success)
180  {
181  client.username = username;
182  client.screen = ScreenFactory.Create(client, server, "TextScreen", "@LoggedIn");
183  client.screen.Show();
184  }
185  else
186  {
187  if (++tries >= maxtries)
188  {
189  client.screen = ScreenFactory.Create(client, server, "Logout");
190  client.screen.Show();
191  }
192  else
193  {
194  LnWrite(catalog.GetString("Username or password incorrect. Please try again."));
195  LnWrite(catalog.GetString("Username") + ": ");
196  status = states.WaitForUsername;
197  }
198  }
199  break;
200  default:
201  break;
202  }
203  }
204  }
