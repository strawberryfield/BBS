100  public override void HandleMessage(string msg)
101  {
102  msg = msg.Trim();
103  if (string.IsNullOrWhiteSpace(msg))
104  {
105  if (status == states.WaitForConfirm) msg = catalog.GetString("Yes");
106  else if (status != states.WaitForUsername) return;
107  }
108  switch (status)
109  {
110  case states.WaitForUsername:
111  string username = msg.ToUpper();
112  bool success = false;
113  switch (username)
114  {
115  case "":
116  client.screen = ScreenFactory.Create(client, server, "LoginScreen");
117  client.screen.Show();
118  break;
119  case "LOGOUT":
120  client.screen = ScreenFactory.Create(client, server, "Logout");
121  client.screen.Show();
122  return;
123  case "GUEST":
124  case "NEW":
125  break;
126  default:
127  if (user.AcceptableUsername(username))
128  using (bbsContext bbs = new bbsContext())
129  {
130  User test = bbs.GetUserByUsername(username);
131  if (test == null) success = true;
132  }
133  break;
134  }
135  if (success)
136  {
137  user.Userid = username;
138  LnWrite(catalog.GetString("Password") + ": ");
139  status = states.WaitForPassword;
140  client.status = EClientStatus.Authenticating;
141  }
142  else
143  {
144  LnWrite(catalog.GetString("This name is already in use. Try another."));
145  LnWrite(catalog.GetString("Username") + ": ");
146  }
147  break;
148 
149  case states.WaitForPassword:
150  password = msg;
151  if (user.AcceptablePassword(password))
152  {
153  LnWrite(catalog.GetString("Confirm password") + ": ");
154  status = states.WaitForConfirmPassword;
155  }
156  else
157  {
158  LnWrite(catalog.GetString("Password do not meet security criteria."));
159  LnWrite(catalog.GetString("Password") + ": ");
160  status = states.WaitForPassword;
161  }
162  break;
163 
164  case states.WaitForConfirmPassword:
165  if (password == msg)
166  {
167  user.SetPassword(password);
168  client.status = EClientStatus.Guest;
169  LnWrite(catalog.GetString("Your real name") + ": ");
170  status = states.WaitForRealName;
171  }
172  else
173  {
174  LnWrite(catalog.GetString("The two passwords do not match. Try again."));
175  LnWrite(catalog.GetString("Password") + ": ");
176  status = states.WaitForPassword;
177  }
178  break;
179 
180  case states.WaitForRealName:
181  user.Realname = msg;
182  LnWrite(catalog.GetString("Your city") + ": ");
183  status = states.WaitForCity;
184  break;
185 
186  case states.WaitForCity:
187  user.City = msg;
188  LnWrite(catalog.GetString("Your country")+": ");
189  status = states.WaitForNation;
190  break;
191 
192  case states.WaitForNation:
193  user.Nation = msg;
194  LnWrite(catalog.GetString("Confirm new user creation? [{0}]/{1}",
195  new object[] {
196  catalog.GetString("Yes"), catalog.GetString("No")
197  }) + ": ");
198  status = states.WaitForConfirm;
199  break;
200 
201  case states.WaitForConfirm:
202  msg = msg.ToUpper();

211  EventLogger.Write(m, client.Remote);
