217  {

424  public override void HandleChar(byte[] data, int bytesReceived)
425  {
426  base.HandleChar(data, bytesReceived);
427  switch (data[0])
428  {
429  case 'N' - 64: // Ctrl-N
430  HandleCursorDown();
431  break;
432  case 'P' - 64: // Ctrl-P
433  HandleCursorUp();
434  break;
435  case 'F' - 64: // Ctrl-F
436  HandlePageDown();
437  break;
438  case 'B' - 64: // Ctrl-B
439  HandlePageUp();
440  break;
441  case 'D' - 64: // Ctrl-D
442  HandleHalfPageDown();
443  break;
444  case 'U' - 64: // Ctrl-U
445  HandleHalfPageUp();
446  break;
447  case 'R' - 64: // Ctrl-R
448  case 'L' - 64: // Ctrl-L
449  HandleRedraw();
450  break;
451  default:
452  break;
453  }

584  public string GetFile(string data)
585  {
586  string assets = ConfigurationManager.AppSettings.Get("assets");
587  if (data.StartsWith('@'))
588  {
589  NameValueCollection texts = (NameValueCollection)ConfigurationManager.GetSection("Texts");
590  data = texts[data.Substring(1)];
591  if (File.Exists(Path.Combine(assets, locale, data)))
592  {
593  data = Path.Combine(locale, data);
594  }
595  }
596 
597  if (TextHelper.IsUrl(data))
598  return data;
599  else
600  return Path.Combine(assets, data);
